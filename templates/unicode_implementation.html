def unicode_encrypt(inputText, shift, multipliers):
    encryptedText = ""
    count = 0
    multipliers = splitMultipliers(multipliers)
    for i in range(len(inputText)):
        charCode = ord(inputText[i])
        shift *= multipliers[count]
        charCode += shift
        charCode = ((charCode - 33) % (1114112 - 33 + 1)) + 33
        encryptedText += chr(charCode)
        shift = shift + 1 if shift > 0 else shift - 1
        count = (count + 1) % len(multipliers)
    return encryptedText
